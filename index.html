Custom RAM IP Core with Byte-Enable

April 2025

Course: Embedded Systems Design I

Designed and implemented a custom VHDL RAM IP core (4K x 32-bit) with byte-enable functionality for Nios II processor integration via Avalon memory-mapped interface in Platform Designer. Developed comprehensive VHDL testbench validating full-word, half-word, and individual byte operations across all 4096 addresses. Verified C integration using polling, interrupts, and peripheral output (LED/UART).

VHDL RAM Design Avalon Interface Nios II Testbench C Platform Designer

Key Features

  • 4096 x 32-bit synchronous RAM with independent byte-lane control
  • Active-low byte-enable signals for selective byte writes
  • Comprehensive testbench with 20,480+ write/read cycles
  • Full integration with Nios II via Avalon MM slave interface
  • Validated memory operations: full-word, half-word, and byte-level access
-------------------------------------------------------------------------
-- Ian Woloshyn
-- CPET 561 Embedded System Design 1
-- Lab 6 raminfr_be.vhd
-- March 26th 2025
-------------------------------------------------------------------------

LIBRARY ieee;
USE ieee.std_logic_1164.ALL;
USE ieee.std_logic_unsigned.ALL;

ENTITY raminfr IS
PORT(
    clk               : IN std_logic;
    reset_n           : IN std_logic;
    writebyteenable_n : IN std_logic_vector(3 DOWNTO 0);
    address           : IN std_logic_vector(11 DOWNTO 0);
    writedata         : IN std_logic_vector(31 DOWNTO 0);
    
    readdata          : OUT std_logic_vector(31 DOWNTO 0)
);
END ENTITY raminfr;

ARCHITECTURE rtl OF raminfr IS
    TYPE ram_type IS ARRAY (4095 DOWNTO 0) OF std_logic_vector (7 DOWNTO 0);
    SIGNAL RAM0 : ram_type;  -- Byte 0
    SIGNAL RAM1 : ram_type;  -- Byte 1
    SIGNAL RAM2 : ram_type;  -- Byte 2
    SIGNAL RAM3 : ram_type;  -- Byte 3
    SIGNAL read_addr : std_logic_vector(11 DOWNTO 0);

BEGIN
    RamBlock : PROCESS(clk)
    BEGIN
        IF (clk'event AND clk = '1') THEN
            IF (reset_n = '0') THEN
                read_addr <= (OTHERS => '0');
                FOR i IN 0 TO 4095 LOOP
                    RAM0(i) <= (OTHERS => '0');
                    RAM1(i) <= (OTHERS => '0');
                    RAM2(i) <= (OTHERS => '0');
                    RAM3(i) <= (OTHERS => '0');
                END LOOP;
            ELSE
                -- Independent byte-enable control
                IF (writebyteenable_n(0) = '0') THEN
                    RAM0(conv_integer(address)) <= writedata(7 DOWNTO 0);
                END IF;
                IF (writebyteenable_n(1) = '0') THEN
                    RAM1(conv_integer(address)) <= writedata(15 DOWNTO 8);
                END IF;
                IF (writebyteenable_n(2) = '0') THEN
                    RAM2(conv_integer(address)) <= writedata(23 DOWNTO 16);
                END IF;
                IF (writebyteenable_n(3) = '0') THEN
                    RAM3(conv_integer(address)) <= writedata(31 DOWNTO 24);
                END IF;
            END IF;
            read_addr <= address;
        END IF;
    END PROCESS RamBlock;

    -- Concurrent read assignment
    readdata(7 DOWNTO 0)   <= RAM0(conv_integer(read_addr));
    readdata(15 DOWNTO 8)  <= RAM1(conv_integer(read_addr));
    readdata(23 DOWNTO 16) <= RAM2(conv_integer(read_addr));
    readdata(31 DOWNTO 24) <= RAM3(conv_integer(read_addr));

END ARCHITECTURE rtl;
-- Testbench validates all byte-enable combinations
stim_proc: process
begin
    reset_n <= '0';
    wait for 20 ns;
    reset_n <= '1';
    wait for 20 ns;

    -- Full Word test (all bytes enabled)
    for i in 0 to 4095 loop
        address <= std_logic_vector(to_unsigned(i, 12));
        writebyteenable_n <= "0000";
        writedata <= x"00000000";
        wait for 20 ns;	
        assert readdata = x"00000000" 
            report "ERROR: Full Word mismatch at address " & integer'image(i);
    end loop;

    -- Top Half Word test
    for i in 0 to 4095 loop
        address <= std_logic_vector(to_unsigned(i, 12));
        writebyteenable_n <= "0011";  -- Enable top bytes
        writedata <= x"11111111";
        wait for 20 ns;
        assert readdata = x"11110000" 
            report "ERROR: Top Half Word mismatch";
    end loop;

    -- Individual byte tests for Bytes 0-3...
    report "Simulation complete - 20,480+ operations validated";
    wait;
end process;
// C Integration with Nios II
#include "io.h"
#include "system.h"

volatile uint32 *ram_base = (uint32 *) RAMINFR_BE_0_BASE;

// Write to custom RAM
void ram_write(uint32 address, uint32 data) {
    *(ram_base + address) = data;
}

// Read from custom RAM
uint32 ram_read(uint32 address) {
    return *(ram_base + address);
}

int main(void) {
    // Test RAM operations
    ram_write(0x000, 0xDEADBEEF);
    uint32 data = ram_read(0x000);
    
    printf("RAM Test: Written 0xDEADBEEF, Read 0x%08X\n", data);
    
    return 0;
}

Servo Motor Controller with PWM & FSM

March 2025

Course: Embedded Systems Design I

Designed and implemented a custom VHDL servo controller IP core with 4-state FSM (SWEEP_LEFT, SWEEP_RIGHT, INT_LEFT, INT_RIGHT), PWM signal generation, and Avalon memory-mapped slave interface. Integrated with Nios II processor using interrupt-driven C handlers for real-time angle control via pushbuttons, with 7-segment display feedback showing min/max angles.

VHDL FSM PWM Nios II Avalon Interface C ISR Platform Designer

Key Features

  • 4-state FSM with automatic sweep between configurable min/max angles
  • 20ms PWM period with 1-2ms pulse width control for servo positioning
  • Interrupt-driven communication with Nios II processor
  • Real-time angle adjustment via pushbutton and switch inputs
  • 7-segment display feedback for visual angle monitoring
-- 4-State FSM for Servo Control
entity lab5 is
    port (
        clk            : in  std_logic;
        reset_n        : in  std_logic;
        write_enable   : in  std_logic;
        write_address  : in  std_logic;
        write_data     : in  std_logic_vector(31 downto 0);
        pwm_out        : out std_logic;
        irq_signal     : out std_logic
    );
end entity Lab5;

architecture Behavioral of Lab5 is
    constant SWEEP_LEFT   : std_logic_vector(1 downto 0) := "00";
    constant SWEEP_RIGHT  : std_logic_vector(1 downto 0) := "01";
    constant INT_LEFT     : std_logic_vector(1 downto 0) := "10";
    constant INT_RIGHT    : std_logic_vector(1 downto 0) := "11";

    signal state_reg      : std_logic_vector(1 downto 0);
    signal period_counter : unsigned(19 downto 0);
    signal angle_count    : unsigned(31 downto 0);
    
begin
    -- Next State Logic
    next_state_proc: process(state_reg, angle_count, min_angle_reg, max_angle_reg)
    begin
        case state_reg is
            when SWEEP_RIGHT =>
                if angle_count >= max_angle_reg then
                    state_next <= INT_RIGHT;  -- Trigger interrupt
                end if;
            when INT_RIGHT =>
                if write_enable = '1' then
                    state_next <= SWEEP_LEFT;
                end if;
            when SWEEP_LEFT =>
                if angle_count <= min_angle_reg then
                    state_next <= INT_LEFT;   -- Trigger interrupt
                end if;
            when INT_LEFT =>
                if write_enable = '1' then
                    state_next <= SWEEP_RIGHT;
                end if;
        end case;
    end process;

    -- PWM Generation
    pwm_proc: process(clk, reset_n)
    begin
        if reset_n = '0' then
            pwm_out <= '0';
        elsif rising_edge(clk) then
            if period_counter < angle_count then
                pwm_out <= '1';
            else
                pwm_out <= '0';
            end if;
        end if;
    end process;
end Behavioral;
// Interrupt Service Routines for Servo Control
volatile uint32 min_angle = 46;
volatile uint32 max_angle = 134;
volatile uint8 update_display = 0;

void pushbutton_ISR(void* context) {
    uint32 button_edge = (*buttons + 3);

    if (button_edge & 0x8) {    // KEY3: Set min angle
        min_angle = *switches;
    }

    if (button_edge & 0x4) {    // KEY2: Set max angle
        max_angle = *switches;
    }

    update_display = 1;
    *(buttons + 3) = 0x00;  // Clear interrupts
}

void servo_ISR(void* context) {
    *(servo + 0) = angleToDooty(min_angle);   // Write min
    *(servo + 1) = angleToDooty(max_angle);   // Write max
}

int angleToDooty(int dooty) {
    int count = dooty * (50000/90) + 50000/2;
    return count;
}

int main(void) {
    // Register ISRs
    alt_ic_isr_register(PUSHBUTTONS_IRQ_INTERRUPT_CONTROLLER_ID, 
                        PUSHBUTTONS_IRQ, pushbutton_ISR, 0, 0);
    alt_ic_isr_register(SERVO_CONTROLLER_0_IRQ_INTERRUPT_CONTROLLER_ID, 
                        SERVO_CONTROLLER_0_IRQ, servo_ISR, 0, 0);

    // Enable interrupts
    *(buttons + 2) |= 0x0C;
    
    while (1) {
        if (update_display) {
            display_angle(min_angle, NULL, hex5, hex4);
            display_angle(max_angle, hex2, hex1, hex0);
            update_display = 0;
        }
    }
    return 0;
}
